    \frametitle{Prädiktoren für Arbeitslosigkeit}
    
    \begin{columns}[T]

    \column{0.5\textwidth}
        \begin{itemize}
            \item Multiples Lineares Modell
            \item[$\rightarrow$] $y = \beta_0 + \beta_1 x_1 + \beta_2 x_2 + \beta_3 x_3$
            \item[$\rightarrow$]{\small y: Erwerbslos, $x_1$: Abitur, \\ $x_2$: OhneAbschl, $x_3$: Erwerbstaetig}
        \end{itemize}
    \begin{lstlisting}
model <- lm(Erwerbslos_Quote ~ 
              Abitur_Quote + 
              OhneAbschl_Quote + 
              Erwerbstaetig_Quote,
            data = df2)

summary(model)
    \end{lstlisting}

    \column{0.5\textwidth}
    \begin{itemize}
    	\item Wichtigste Erkenntnisse:
    \end{itemize}
\begin{lstlisting}
(Intercept)	      +9.88   ***
OhneAbschl_Quote      +0.22   ***
Erwerbstaetig_Quote   -0.11   ***
Abitur_Quote          +0.01   .

R^2 = 0.34
p-value: < 2.2e-16
\end{lstlisting}

    \begin{itemize}
        \item[$\rightarrow$] $y = 9.88 + 0.22 x_1 - 0.11 x_2 + 0.01 x_3$
        \item[$\rightarrow$] Ohne Abschluss Quote wichtigster Prädiktor
    \end{itemize}

    \end{columns}
\end{frame}
